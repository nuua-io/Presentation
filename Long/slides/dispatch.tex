\begin{slide}
    \slidecaps{PROBLEM}
    \begin{itemize}
        \item Threaded dispatch is not ANSI C compilant.
    \end{itemize}
    \slidecaps{SOLUTION}
    \begin{itemize}
        \item Use a switch dispatch
    \end{itemize}
    \slidecaps{TRADEOFF}
    \begin{itemize}
        \item ANSI C compilant.
        \item Less efficient than a threaded dispatch.
        \item Simple to implement.
    \end{itemize}
\end{slide}
